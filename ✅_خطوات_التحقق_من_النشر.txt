========================================
   ✅ خطوات التحقق من النشر
========================================

اتبع هذه الخطوات بالترتيب للتحقق من نجاح النشر:

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الخطوة 1️⃣: تحقق من حالة النشر في Render
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. اذهب إلى: https://dashboard.render.com

2. سجل الدخول بحسابك

3. افتح خدمة "Backend" أو "ecom-parent-project"

4. انظر إلى قسم "Events" أو "Logs"

5. تحقق من:
   ✅ Status = "Live" (أخضر)
   ❌ إذا كان "Deploy failed" (أحمر) - انتقل إلى الخطوة 6

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الخطوة 2️⃣: اختبر لوحة الإدارة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. افتح: https://ecom-parent-project.onrender.com/admin/

2. سجل الدخول بحساب الإدارة

3. ابحث عن قسمين جديدين:
   ✅ "الطلبات الجديدة" (New Orders)
   ✅ "الطلبات المعالجة" (Processed Orders)

4. إذا لم تجدهما:
   - اضغط Ctrl+Shift+R لمسح الكاش
   - أعد تحميل الصفحة
   - إذا لم تظهر، انتقل إلى الخطوة 6

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الخطوة 3️⃣: اختبر الطلبات الجديدة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. افتح قسم "الطلبات الجديدة"

2. يجب أن ترى فقط الطلبات بحالة "في الانتظار" (pending)

3. اختر طلب واحد

4. من قائمة Actions، اختر "✅ تأكيد الطلبات المحددة"

5. اضغط "Go"

6. يجب أن ترى رسالة: "تم تأكيد 1 طلب وتم نقلهم إلى قسم الطلبات المعالجة"

7. الطلب يجب أن يختفي من قائمة "الطلبات الجديدة"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الخطوة 4️⃣: اختبر الطلبات المعالجة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. افتح قسم "الطلبات المعالجة"

2. يجب أن ترى الطلب الذي أكدته في الخطوة 3

3. يجب أن ترى جميع الطلبات ما عدا "في الانتظار"

4. اختر طلب بحالة "مؤكد" (confirmed)

5. من قائمة Actions، اختر "⚙️ تحديد كـ قيد التحضير"

6. اضغط "Go"

7. يجب أن تتغير حالة الطلب إلى "قيد التحضير" (preparing)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الخطوة 5️⃣: اختبر حساب الخصم في الموقع
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. افتح الموقع: https://ecom-parent-project.onrender.com/

2. اختر منتج سعره 30,000 IQD

3. أضفه إلى السلة

4. افتح السلة

5. تحقق من:
   ✅ السعر الأصلي: 30,000 IQD
   ✅ الخصم: -2,000 IQD
   ✅ السعر بعد الخصم: 28,000 IQD

6. أكمل الطلب

7. افتح لوحة الإدارة

8. افتح الطلب الجديد

9. تحقق من:
   ✅ السعر المحفوظ = 28,000 IQD (السعر المخصوم)
   ✅ ليس 30,000 IQD

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الخطوة 6️⃣: إذا ظهرت مشاكل
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. اذهب إلى Render Dashboard

2. افتح خدمة Backend

3. اضغط على "Logs"

4. ابحث عن رسائل الخطأ (باللون الأحمر)

5. انسخ رسالة الخطأ كاملة

6. أرسلها لي مع:
   - وقت حدوث الخطأ
   - ماذا كنت تفعل عندما حدث الخطأ
   - لقطة شاشة إن أمكن

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ إذا نجحت جميع الخطوات:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ استخدام حساب الخصم الصحيح
✅ حفظ السعر المخصوم في الطلبات
✅ تحديث حالة الطلبات بسهولة
